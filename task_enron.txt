db.enron.aggregate([
    {$unwind : "$headers.To" },
    {$group:{_id:{from:"$headers.From", to:"$headers.To"},count:{$sum:1}}},
    {$sort:{"count":-1}},
    {$limit:1}
    ],
    {allowDiskUse: true}
)
